USAGE EXAMPLE

from formulas.renaissance_compounding import RenaissanceMasterController

# Create system for $100 -> $10,000
system = RenaissanceMasterController(
    initial_capital=100.0,
    target_capital=10000.0
)

# Update with market data
system.update(price=97500.0, volume=1.5)

# Record trade results
system.record_trade(pnl=0.40, new_capital=100.40, timestamp=time.time())

# Check if should trade
if system.should_trade():
    position_size = system.get_position_size()
    print(f"Trade ${position_size:.2f}")

# Get full status
status = system.get_full_status()
print(f"Progress: {status['capital']['progress_pct']:.1f}%")
print(f"Days to target: {status['time_to_target'][1]:.1f}")
